\documentclass{article}
%%%<
\usepackage{verbatim}
%%%>

\usepackage{tikz}   %TikZ is required for this to work.  Make sure this exists before the next line

\usepackage{tikz-3dplot} %requires 3dplot.sty to be in same directory, or in your LaTeX installation
\usepackage{tkz-euclide}

\usetkzobj{all}
\usetikzlibrary{calc}
\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work
\PreviewEnvironment{tikzpicture}
\setlength\PreviewBorder{2mm}

\begin{document}
\begin{tikzpicture}

\tkzDefPoint(0,0){O}
\tkzDefPoint(-2,-4){x}
\tkzDefPoint(4,-5){z}
\tkzDefPoint(1, -3){y}

\draw[->] (O) -- (x) node[anchor=north east]{$x$};
\draw[->] (O) -- (z) node[anchor=north west]{$z$};
\draw[->] (O) -- (y) node[anchor=north west]{$\hat y$};

\begin{pgfinterruptboundingbox} % remove blank space
\tkzDefLine[parallel=through y](z,O);
\tkzGetPoint{py1}
\tkzInterLL(O,x)(py1,y);
\tkzGetPoint{betax}
\draw[dashed] (y) -- (betax);
\draw[->, dashed] (O) -- (betax) node[anchor=east]{$\hat{\beta}_{1}^{(1)}x$};

\tkzDefLine[parallel=through y](x,O);
\tkzGetPoint{py2}
\tkzInterLL(O,z)(py2,y);
\tkzGetPoint{betaz}
\draw[dashed] (y) -- (betaz);
\draw[->, dashed] (O) -- (betaz) node[anchor=south west]{$\hat{\beta}_{2}^{(1)}z$};
\end{pgfinterruptboundingbox}
\end{tikzpicture}
\end{document}
