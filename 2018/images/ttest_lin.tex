\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work

\begin{document}
\begin{tikzpicture}

\tkzDefPoint(0,0){O}
\tkzDefPoint(-2,-3){x}
\tkzDefPoint(3,-4){y}

\draw[->] (O) -- (x) node[anchor=north east]{$x^c$};
\draw[->] (O) -- (y) node[anchor=north west]{$y^c$};
\tkzMarkAngle[size=0.4](x,O,y)
\tkzLabelAngle[pos=0.6](x,O,y){$\varphi$}

\begin{pgfinterruptboundingbox}
\tkzDefLine[perpendicular=through y, K=-.3](x,O);
\tkzGetPoint{hy}
\tkzDefPointBy[projection=onto x--O](hy)
\tkzGetPoint{h}
\tkzMarkRightAngle[size=0.15](x,h,y)
\draw[->] (h) -- (y) node[midway,below,rotate=-33]{\footnotesize$\sqrt{RSS}$};
\draw[->] (O) -- (h) node[anchor=east]{$\hat{\beta}_{2} x^c$};
\end{pgfinterruptboundingbox}

\end{tikzpicture}
