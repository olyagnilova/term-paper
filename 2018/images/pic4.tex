\documentclass{article}
%%%<
\usepackage{verbatim}
%%%>

\usepackage{tikz}   %TikZ is required for this to work.  Make sure this exists before the next line

\usepackage{tikz-3dplot} %requires 3dplot.sty to be in same directory, or in your LaTeX installation
\usepackage{tkz-euclide}
\usetkzobj{all}
\usetikzlibrary{calc}
\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work
\PreviewEnvironment{tikzpicture}
\setlength\PreviewBorder{2mm}

\begin{document}
\begin{tikzpicture}

\tkzDefPoint(0,0){O}
\tkzDefPoint(-1,-3){x}
\tkzDefPoint(4,-6){y}

\draw[->] (O) -- (x) node[anchor=north east]{$x^c$};
\draw[->] (O) -- (y) node[anchor=north west]{$y^c$};
\tkzMarkAngle[size=0.4](x,O,y)
\tkzLabelAngle[pos=0.6](x,O,y){$\varphi$}

\tkzDefLine[perpendicular=through y, K=-.3](x,O);
\tkzGetPoint{hy}
\tkzDefPointBy[projection=onto x--O](hy)
\tkzGetPoint{h}
\tkzMarkRightAngle(x,h,y)
\draw[dashed] (y) -- (h);
\draw[->, dashed] (O) -- (h) node[anchor=north east]{$\hat{\beta}_{yx}x^c$};

\tkzDefLine[perpendicular= through x, K=-.9](y,O);
\tkzGetPoint{hx}
\tkzDefPointBy[projection=onto y--O](hx)
\tkzGetPoint{p}
\tkzMarkRightAngle(O,p,x)
\draw[dashed] (x) -- (p);
\draw[->, dashed] (O) -- (p) node[anchor=west]{$\hat{\beta}_{xy}y^c$};

\end{tikzpicture}
\end{document}
